178void ClientApp::ProcessMessage(const Message& msg)
179{
180 // Tries to send the message as it comes, if not possible, then adds to pending message list
181 try
182 {
183 // If message is a text message
184 const TextMessage& text = dynamic_cast<const TextMessage&>(msg);
185
186 uint64_t recipientHash = text.GetRecipientHash();
187
188 try
189 {
190 ntwk::Socket& recipientSocket = ::GLOBAL_SERVER.GetClientSockFromClientHash(recipientHash);
191
192 std::wstring content;
193 text.Serialize(content);
194
195 // Send the message if the client socket is valid
196 if (m_Socket == INVALID_SOCKET)
197 {
198 Logger::logfmt<Log::ERR>("Could not send message to Recipient#ld, because Sender's socket (Client#%ld) is an \"INVALID_SOCKET\"", text.GetRecipientHash(), text.GetSenderHash());
199 return;
200 }
201 else
202 {
203 auto recipientApp = ::GLOBAL_SERVER.GetClientAppFromHash(recipientHash);
204 *recipientApp << content;
205 // Raise a MSG_DELIVERED event
206 Event<EventName::MSG_DELIVERED>::Raise(std::make_shared<TextMessage>(text.GetSenderHash(), text.GetRecipientHash(), text.GetContent()));
207 }
208
209 }
210 catch (const std::out_of_range& e)
211 {
212 Logger::logfmt<Log::ERR, 2048>("Recipient #%ld was not found, message not sent. Adding to pending message list: %s", recipientHash, e.what());
213 m_PendingMessages.push_back(std::make_shared<TextMessage>(text));
214 }
215 }
216 catch (std::bad_cast& e)
217 {
218 // Do nothing
219 }
220
221 // When nothing matches
222 //throw std::runtime_error("Unidentified message type passed.");
223}
